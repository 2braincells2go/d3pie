Standalone Examples » Tooltips #2

This shows how to display auto-generated tooltips on your chart that contains whatever segment info you want (label, value, percent, or a combination). 

var pie = new d3pie("pie", {
  header: {
    title: {
      text: "Tooltips: example #1",
      fontSize: 30
    }
  },
  labels: {
    inner: {
      format: "none"
    }
  },
  data: {
    content: [
      { label: "JavaScript", value: 1, caption: "JS Rocks! Woo!" },
      { label: "Ruby", value: 2, caption: "Ruby is foreign and weird" },
      { label: "Java", value: 3, caption: "Old and verbose (buuuuurn!)"}
    ]
  },
  tooltips: {
    enabled: true,
    type: "caption"
  }
});
